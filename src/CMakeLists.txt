PROJECT (ziahttpd)

SET(ZIAHTTPD_SRCS main.cpp)
SET(ZIAHTTPD_HEADERS zia.h singleton.hpp ISocket.hpp zSocket_win32.h)

IF (WIN32)
	SET(ZIAHTTPD_LIBS Ws2_32.lib)
	SET(ZIAHTTPD_SRCS ${ZIAHTTPD_HEADERS} ${ZIAHTTPD_SRCS} zSocket_win32.cpp)
ENDIF (WIN32)

IF (UNIX)
  SET(ZIAHTTPD_SRCS ${ZIAHTTPD_SRCS})
ENDIF (UNIX)

ADD_EXECUTABLE (ziahttpd ${ZIAHTTPD_SRCS})
TARGET_LINK_LIBRARIES(ziahttpd ${ZIAHTTPD_LIBS})
